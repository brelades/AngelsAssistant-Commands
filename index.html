} /* ========================= EXCLUDED COMMANDS ========================= */ const EXCLUDED = new Set([ "addpoints", "removepoints", "restock", "giveaway", "points-add", "points-remove", "disable", "enable", "disabled-commands-list" ]); } } /* ========================= SORT ========================= */ grouped.Twitch.sort((a, b) => a.name.localeCompare(b.name)); grouped.Discord.sort((a, b) => a.name.localeCompare(b.name)); /* ========================= HTML ========================= */ const html = ` AngelsAssistant Dashboard
